def fetch(uri, limit = 6)
  response = Net::HTTP.get_response(uri)

  case response.code.to_i
  when 200
    puts response.body
  when 300, 301
    unless limit.zero?
      fetch URI.parse(response['Location']), limit.pred
    else
      response.value
    end
  else
    response.value
  end
end

def what_class(obj)
  case obj
  when proc { |x| x.kind_of? String }
    String
  when proc { |x| x.kind_of? Numeric }
    Numeric
  else
    Class
  end
end
